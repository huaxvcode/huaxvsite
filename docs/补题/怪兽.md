```cpp
#include <bits/stdc++.h>
#define long long long
#define bint __int128

long exgcd(long a, long b, long& x, long& y) {
    if (b == 0) { x = 1; y = 0; return a; }
    long d = exgcd(b, a % b, y, x);
    y -= a / b * x;
    return d;
}

long p, q, n1, n2, n3;
#define can_not {std::cout << -1 << '\n'; return;}
#define debug {std::cout << "debug here\n";}

void solve() {
    std::cin >> p >> q >> n1 >> n2 >> n3;
    long a = n3 - n1, b = n3 - n2, c = p * n3 - q;
    if (p * n3 < q) {
        can_not;
    }
    if (a == 0) { // 全部相同
        if (c) can_not;
        std::cout << 0 << ' ' << p << '\n';
        return;
    }
    else if (b == 0) { // 最大的两个相同
        long t = (p * n3 - q) / (n3 - n1);
        if (t * (n3 - n1) != p * n3 - q) can_not;
        if (t < 0 || t > p) can_not;
        std::cout << 0 << ' ' << p - t << '\n';
        return;
    }
    else {
        long d, x, y;
        d = exgcd(a, b, x, y);
        if (c % d) can_not;
        a /= d; b /= d; c /= d;
        x *= c; y *= c;
        long t = (x % b + b) % b;
        t = (t - x) / b;
        // 求出 x 的最小正整数解
        long xx = x + t * b, yy = y - t * a;
        // std::cout << xx << ' ' << yy << '\n';
        // x 增，则 y 减；y 增，则 x 减
        if (yy < 0) can_not;
        // 求出 y 的最小正整数，此时是 x 的最大正整数解
        t = (y % a + a) % a;
        t = (y - t) / a;
        long xs = x + b * t, ys = y - a * t;
        t = (xs - xx) / b; 
        long a_ = (a - b), b_ = p - xx - yy;
        if (a_ == 0) {
            if (b_ >= 0) std::cout << b_ << ' ' << b_ << '\n';
            else can_not; 
        }
        // 求出 p - x' - y' 的最小值和最大值，其中 x'、y' 是 x、y 的通解
        // xx <= x' <= xs
        // 0 <= p - x' - y' <= p
        else if (a_ > 0) {
            long mi = (b_ % a_ + a_) % a_;
            long mit = (mi - b_) / a_;
            mit = std::min(mit, t);
            long ans_mi, ans_mx;
            if (mit < 0) ans_mi = b_;
            else ans_mi = mit * a_ + b_;
            long t_ = (xx + yy) / a;
            t_ = std::min(t_, t);
            ans_mx = a_ * t_ + b_;
            if (ans_mi > p || ans_mi < 0 || ans_mx > p || ans_mx < 0) can_not;
            std::cout << ans_mi << ' ' << ans_mx << '\n';
        }
        else {
            long fa_ = -a_;
            long mi = (b_ % fa_ + fa_) % fa_;
            long mxt = (b_ - mi) / fa_;
            mxt = std::min(mxt, t);
            long ans_mi = a_ * mxt + b_, ans_mx;
            ans_mx = b_;
            if (ans_mi > p || ans_mi < 0 || ans_mx > p || ans_mx < 0) can_not;
            std::cout << ans_mi << ' ' << ans_mx << '\n';
        }
    }
    
}

int main() {
    std::ios::sync_with_stdio(0);
    std::cin.tie(0); std::cout.tie(0);
    
    solve();
    return 0;
}
```
