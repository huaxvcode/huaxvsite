#include <bits/stdc++.h>
typedef long long var;
typedef __int128 vhh;

namespace {



void solve() {
    std::cout << sizeof(vhh) << '\n';   // 16
    std::cout << sizeof(var) << '\n';   // 8
    std::cout << sizeof(int) << '\n';   // 4
    std::cout << RAND_MAX << '\n';      // 2147483647
    auto dfs = [&](auto&& f, auto&& x) -> void {
        if (x >= 0) {
            std::cout << x << '\n';
            f(f, x - 1);
        }
    };
    dfs(dfs, 100);
}

}

// g++ -std=c++20 Main.cpp -o Main
int main() {
    std::ios::sync_with_stdio(0);
    std::cin.tie(0); std::cout.tie(0);
    std::random_device rd;
    std::mt19937 mt(rd());
    srand(mt());
    int t = 1; 
    // std::cin >> t;
    for (int i = 1; i <= t; i ++) {
        // std::cout << "Case #1: ";
        solve();
    }
    return 0;
}
